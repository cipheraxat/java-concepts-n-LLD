Stream pipeline: Source → Intermediate ops (lazy) → Terminal op (eager, triggers execution)

◎ SDE2 Interview Focus
  • "Can you reuse a stream?" => NO — streams are single-use. Throws IllegalStateException.
  • "What is the difference between map and flatMap?" => map: 1-to-1; flatMap: 1-to-0..n (flattens Stream<Stream<T>> to Stream<T>)
  • "When would you use reduce?" => Aggregate: sum, product, max, custom accumulation
  • "parallelStream() pitfalls?" => Stateful ops (sorted), shared mutable state, small datasets
  • "Collectors.groupingBy vs partitioningBy?" => groupingBy: group by key (Map<K, List<T>>); partitioningBy: split into true/false groups

package lessons.java8_features;

import java.util.*;
import java.util.stream.*;


/**
 * LESSON 06B — Stream API (Most Important Java 8 Feature for Interviews)
 *
 * Stream pipeline: Source → Intermediate ops (lazy) → Terminal op (eager, triggers execution)
 *
 * SDE2 Interview Questions:
 *  - "Can you reuse a stream?" => NO — streams are single-use. Throws IllegalStateException.
 *  - "What is the difference between map and flatMap?"
 *    => map: 1-to-1; flatMap: 1-to-0..n (flattens Stream<Stream<T>> to Stream<T>)
 *  - "When would you use reduce?" => Aggregate: sum, product, max, custom accumulation
 *  - "parallelStream() pitfalls?" => Stateful ops (sorted), shared mutable state, small datasets
 *  - "Collectors.groupingBy vs partitioningBy?"
 *    => groupingBy: group by key (Map<K, List<T>>); partitioningBy: split into true/false groups
 */
public class StreamAPIDemo {

    record Employee(String name, String dept, double salary, int age) {}

    static List<Employee> employees() {
        return List.of(
            new Employee("Alice",   "Engineering", 95000, 32),
            new Employee("Bob",     "Engineering", 87000, 28),
            new Employee("Charlie", "Marketing",   72000, 35),
            new Employee("Dave",    "Engineering",105000, 40),
            new Employee("Eve",     "HR",          65000, 27),
            new Employee("Frank",   "Marketing",   78000, 31),
            new Employee("Grace",   "HR",          70000, 29)
        );
    }

    public static void main(String[] args) {
        List<Employee> emps = employees();

        // ════════════════════════════════════════════════════════════════════════
        // SECTION 1 — SOURCE
        // How streams are created. No processing happens here.
        // ════════════════════════════════════════════════════════════════════════

        int[] arr = {5, 3, 1, 4, 2};

        System.out.println("=== Sources ===");
        // Collection.stream() — most common; sequential by default
        System.out.println("Collection.stream(): " + emps.stream().count() + " employees");

        // Stream.of() — from individual values
        System.out.print("Stream.of():          "); Stream.of("Alice", "Bob", "Charlie").forEach(s -> System.out.print(s + " ")); System.out.println();

        // Arrays.stream() — from a primitive or object array
        System.out.println("Arrays.stream(arr) sum: " + Arrays.stream(arr).sum());

        // IntStream.range / rangeClosed — primitive int sequences (no boxing)
        System.out.print("range(0,5):       "); IntStream.range(0, 5).forEach(i -> System.out.print(i + " ")); System.out.println();
        System.out.print("rangeClosed(1,5): "); IntStream.rangeClosed(1, 5).forEach(i -> System.out.print(i + " ")); System.out.println();

        // Stream.iterate / Stream.generate — infinite; must pair with limit()
        System.out.print("iterate x2:       "); Stream.iterate(1, n -> n * 2).limit(5).forEach(n -> System.out.print(n + " ")); System.out.println();
        System.out.print("generate randoms: "); Stream.generate(Math::random).limit(3).map(d -> String.format("%.2f", d)).forEach(s -> System.out.print(s + " ")); System.out.println();


        // ════════════════════════════════════════════════════════════════════════
        // SECTION 2 — INTERMEDIATE OPERATIONS
        // Lazy — they transform the stream but do NOT execute until a terminal op
        // is called. Each returns a new Stream.
        //
        //  filter(Predicate)        — keep elements matching condition
        //  map(Function)            — 1-to-1 transform each element
        //  flatMap(Function)        — 1-to-many, then flatten into one stream
        //  sorted() / sorted(Comp)  — natural or custom order
        //  distinct()               — remove duplicates (uses equals/hashCode)
        //  limit(n)                 — keep only first n elements
        //  skip(n)                  — discard first n elements
        //  peek(Consumer)           — side-effect for debugging; does NOT consume
        // ════════════════════════════════════════════════════════════════════════

        // ── filter + map + sorted ──────────────────────────────────────────────
        System.out.println("\n=== filter + map + sorted ===");
        List<String> engNames = emps.stream()
            .filter(e -> e.dept().equals("Engineering"))  // intermediate: lazy
            .map(Employee::name)                           // intermediate: lazy
            .sorted()                                      // intermediate: lazy
            .collect(Collectors.toList());                 // terminal: triggers execution
        System.out.println("Engineering employees: " + engNames);

        // ── distinct + limit + skip ────────────────────────────────────────────
        System.out.println("\n=== distinct + limit + skip ===");
        List<String> depts = emps.stream()
            .map(Employee::dept)
            .distinct()   // removes duplicate department names
            .sorted()
            .collect(Collectors.toList());
        System.out.println("Unique departments: " + depts);

        List<Employee> page = emps.stream()
            .sorted(Comparator.comparing(Employee::name))
            .skip(2)    // discard first 2 (pagination offset)
            .limit(3)   // take next 3   (page size)
            .collect(Collectors.toList());
        System.out.println("Page(skip=2, limit=3): " + page.stream().map(Employee::name).toList());

        // ── flatMap ────────────────────────────────────────────────────────────
        System.out.println("\n=== flatMap ===");
        List<List<Integer>> nested = List.of(List.of(1,2,3), List.of(4,5), List.of(6,7,8,9));
        List<Integer> flat = nested.stream()
            .flatMap(Collection::stream)  // Stream<List<Integer>> → Stream<Integer>
            .collect(Collectors.toList());
        System.out.println("Flattened: " + flat);

        // Practical: unique words from a list of sentences
        List<String> sentences = List.of("hello world", "java streams", "hello java");
        List<String> words = sentences.stream()
            .flatMap(s -> Arrays.stream(s.split(" ")))  // split each sentence into words
            .distinct()
            .sorted()
            .collect(Collectors.toList());
        System.out.println("Unique words: " + words);

        // ── peek (debug only — never rely on it for side effects) ──────────────
        System.out.println("\n=== peek (debug) ===");
        long peekedCount = emps.stream()
            .filter(e -> e.salary() > 80000)
            .peek(e -> System.out.println("  [peek] processing: " + e.name()))
            .count();
        System.out.println("  Count earning >80k: " + peekedCount);

        // ── mapToDouble / mapToInt (primitive specialisations) ─────────────────
        System.out.println("\n=== mapToDouble / mapToInt ===");
        double[] salaries = emps.stream()
            .mapToDouble(Employee::salary)  // Stream<Employee> → DoubleStream (no boxing)
            .toArray();
        System.out.println("Salary array length: " + salaries.length);

        int[] ages = emps.stream()
            .mapToInt(Employee::age)        // Stream<Employee> → IntStream
            .toArray();
        System.out.println("Age array: " + Arrays.toString(ages));


        // ════════════════════════════════════════════════════════════════════════
        // SECTION 3 — TERMINAL OPERATIONS
        // Eager — trigger the stream pipeline and produce a result or side-effect.
        // After a terminal op the stream is consumed and cannot be reused.
        //
        //  collect(Collector)           — gather into collection / map / string
        //  reduce(identity, BinaryOp)   — fold elements into a single value
        //  forEach(Consumer)            — iterate for side-effects
        //  count()                      — number of elements
        //  min(Comparator)              — smallest element
        //  max(Comparator)              — largest element
        //  sum() / average()            — primitive stream shortcuts
        //  anyMatch / allMatch / noneMatch — short-circuit boolean checks
        //  findFirst() / findAny()      — return first / any element (Optional)
        // ════════════════════════════════════════════════════════════════════════

        // ── reduce ────────────────────────────────────────────────────────────
        System.out.println("\n=== reduce / sum / average ===");
        double totalSalary = emps.stream()
            .mapToDouble(Employee::salary)
            .sum();   // terminal: sum is a shortcut for reduce(0, Double::sum)
        System.out.println("Total salary: " + totalSalary);

        OptionalDouble avgSalary = emps.stream()
            .mapToDouble(Employee::salary)
            .average(); // terminal
        System.out.println("Avg salary: " + avgSalary.orElse(0));

        // Custom reduce: compute factorial — reduce(identity, accumulator)
        int factorial = IntStream.rangeClosed(1, 5)
            .reduce(1, (a, b) -> a * b);  // 1*1*2*3*4*5
        System.out.println("5! = " + factorial);

        // Array sum via terminal
        int arraySum = Arrays.stream(arr).sum();
        System.out.println("Array sum: " + arraySum);

        // ── min / max ─────────────────────────────────────────────────────────
        System.out.println("\n=== min / max ===");
        Employee richest = emps.stream()
            .max(Comparator.comparingDouble(Employee::salary))
            .orElseThrow();  // throws NoSuchElementException if stream is empty
        System.out.println("Highest earner: " + richest.name() + " $" + richest.salary());

        Employee youngest = emps.stream()
            .min(Comparator.comparingInt(Employee::age))
            .orElseThrow();
        System.out.println("Youngest: " + youngest.name() + " age " + youngest.age());

        // ── count ─────────────────────────────────────────────────────────────
        System.out.println("\n=== count ===");
        long highEarners = emps.stream()
            .filter(e -> e.salary() > 80000)
            .count();  // terminal
        System.out.println("Earning >80k: " + highEarners);

        // ── anyMatch / allMatch / noneMatch (short-circuit) ───────────────────
        System.out.println("\n=== anyMatch / allMatch / noneMatch ===");
        System.out.println("Any HR?      " + emps.stream().anyMatch(e -> e.dept().equals("HR")));
        System.out.println("All adults?  " + emps.stream().allMatch(e -> e.age() >= 18));
        System.out.println("None Sales?  " + emps.stream().noneMatch(e -> e.dept().equals("Sales")));

        // ── findFirst / findAny ───────────────────────────────────────────────
        System.out.println("\n=== findFirst / findAny ===");
        Optional<Employee> first = emps.stream()
            .filter(e -> e.dept().equals("HR"))
            .findFirst();  // deterministic — first in encounter order
        first.ifPresent(e -> System.out.println("First HR employee: " + e.name()));

        Optional<Employee> any = emps.parallelStream()
            .filter(e -> e.dept().equals("Marketing"))
            .findAny();    // non-deterministic in parallel — whichever thread finds first
        any.ifPresent(e -> System.out.println("Any Marketing employee: " + e.name()));

        // ── forEach ───────────────────────────────────────────────────────────
        System.out.println("\n=== forEach ===");
        System.out.print("Names (forEach): ");
        emps.stream()
            .map(Employee::name)
            .limit(3)
            .forEach(name -> System.out.print(name + " "));
        System.out.println();


        // ════════════════════════════════════════════════════════════════════════
        // SECTION 4 — PARALLEL STREAMS
        // Splits work across threads in ForkJoinPool.commonPool().
        // Use when: large data, CPU-bound, stateless, order doesn't matter.
        // Avoid when: small data, stateful ops, shared mutable state, I/O bound.
        // ════════════════════════════════════════════════════════════════════════

        System.out.println("\n=== parallelStream — count ===");
        long parallelCount = emps.parallelStream()   // splits work across CPU cores
            .filter(e -> e.salary() > 70000)
            .count();
        System.out.println("Parallel count (>70k): " + parallelCount);

        System.out.println("\n=== parallelStream — sum ===");
        double parallelTotalSalary = emps.parallelStream()
            .mapToDouble(Employee::salary)
            .sum();   // sum is safe for parallel (associative operation)
        System.out.println("Parallel total salary: " + parallelTotalSalary);

        // WARNING: avoid shared mutable state in parallel streams
        // BAD example (race condition):
        //   List<String> shared = new ArrayList<>();
        //   emps.parallelStream().forEach(e -> shared.add(e.name())); // NOT thread-safe!
        // GOOD: use collect() instead — it handles thread safety internally
        List<String> safeNames = emps.parallelStream()
            .map(Employee::name)
            .collect(Collectors.toList());  // safe
        System.out.println("Safe parallel collect: " + safeNames.size() + " names");


        // ════════════════════════════════════════════════════════════════════════
        // SECTION 5 — COLLECTORS
        // Collectors are strategies passed to collect() to accumulate stream
        // elements into a result container (List, Map, String, etc.)
        //
        //  toList() / toSet() / toUnmodifiableList()
        //  joining(delimiter, prefix, suffix)
        //  groupingBy(classifier)           — Map<K, List<T>>
        //  groupingBy(classifier, downstream) — with a secondary collector
        //  partitioningBy(predicate)        — Map<Boolean, List<T>>
        //  toMap(keyMapper, valueMapper)
        //  counting()                       — downstream: count per group
        //  averagingDouble / summingDouble  — downstream: numeric aggregation
        //  teeing(c1, c2, merger)           — Java 12+: apply two collectors at once
        // ════════════════════════════════════════════════════════════════════════

        // ── toList / toSet ─────────────────────────────────────────────────────
        System.out.println("\n=== toList / toSet ===");
        List<String> nameList = emps.stream().map(Employee::name).collect(Collectors.toList());
        Set<String>  deptSet  = emps.stream().map(Employee::dept).collect(Collectors.toSet());
        System.out.println("Name list: " + nameList);
        System.out.println("Dept set:  " + deptSet);

        // ── joining ───────────────────────────────────────────────────────────
        System.out.println("\n=== joining ===");
        String namesCsv = emps.stream()
            .map(Employee::name)
            .sorted()
            .collect(Collectors.joining(", ", "[", "]")); // delimiter, prefix, suffix
        System.out.println("All names: " + namesCsv);

        // ── toMap ─────────────────────────────────────────────────────────────
        System.out.println("\n=== toMap ===");
        Map<String, Double> nameToSalary = emps.stream()
            .collect(Collectors.toMap(Employee::name, Employee::salary));
        System.out.println("Alice salary: " + nameToSalary.get("Alice"));

        // ── groupingBy ────────────────────────────────────────────────────────
        System.out.println("\n=== groupingBy ===");
        Map<String, List<Employee>> byDept = emps.stream()
            .collect(Collectors.groupingBy(Employee::dept));  // key = dept name
        byDept.forEach((dept, list) ->
            System.out.println("  " + dept + ": " + list.stream().map(Employee::name).toList()));

        // groupingBy with downstream collectors
        System.out.println("\n=== groupingBy + downstream (averagingDouble / counting) ===");
        Map<String, Double> avgByDept = emps.stream()
            .collect(Collectors.groupingBy(Employee::dept,
                     Collectors.averagingDouble(Employee::salary))); // avg salary per dept
        avgByDept.forEach((d, avg) -> System.out.printf("  %s avg salary: %.0f%n", d, avg));

        Map<String, Long> countByDept = emps.stream()
            .collect(Collectors.groupingBy(Employee::dept, Collectors.counting())); // headcount
        System.out.println("  Headcount: " + countByDept);

        // ── partitioningBy ────────────────────────────────────────────────────
        System.out.println("\n=== partitioningBy ===");
        Map<Boolean, List<Employee>> seniorJunior = emps.stream()
            .collect(Collectors.partitioningBy(e -> e.salary() >= 80000)); // true / false buckets
        System.out.println("Senior (>=80k): " + seniorJunior.get(true).stream().map(Employee::name).toList());
        System.out.println("Junior (<80k):  " + seniorJunior.get(false).stream().map(Employee::name).toList());

        // ── teeing (Java 12+) ─────────────────────────────────────────────────
        System.out.println("\n=== Collectors.teeing (Java 12+) ===");
        // Runs two collectors in one pass, then merges results — avoids two full iterations
        record Stats(long count, double avg) {}
        Stats stats = emps.stream()
            .collect(Collectors.teeing(
                Collectors.counting(),
                Collectors.averagingDouble(Employee::salary),
                Stats::new  // merger: (Long, Double) -> Stats
            ));
        System.out.println("Count: " + stats.count() + ", Avg salary: " + stats.avg());
    }
}
